LOAD DATA INFILE 'C:\\ProgramData\\MySQL\\MySQL Server 9.4\\Uploads\\restaurantes.csv'
INTO TABLE restaurantes
CHARACTER SET utf8mb4
FIELDS TERMINATED BY ',' 
ENCLOSED BY '"'
LINES TERMINATED BY '\n'
IGNORE 1 ROWS
(@ID_RESTAURANTE_csv, @NOMBRE_csv, @CIUDAD_csv, @CCAA_csv, @TIPO_COMIDA_csv, @PRESUPUESTO_csv, @ESTRELLA_MICH_csv, @CADENA_csv)
SET
    ID_RESTAURANTE = REPLACE(@ID_RESTAURANTE_csv, '\r', ''),
    NOMBRE = REPLACE(@NOMBRE_csv, '\r', ''),
    CIUDAD = REPLACE(@CIUDAD_csv, '\r', ''),
    CCAA = REPLACE(@CCAA_csv, '\r', ''),
    T_COMIDA = REPLACE(@TIPO_COMIDA_csv, '\r', ''),
    PRESUPUESTO = REPLACE(@PRESUPUESTO_csv, '\r', ''),
    ESTRELLA_MICH = REPLACE(@ESTRELLA_MICH_csv, '\r', ''),
    CADENA = REPLACE(@CADENA_csv, '\r', '');
